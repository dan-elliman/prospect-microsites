# Two Sigma: self-managed KubeRay cluster config
apiVersion: ray.io/v1alpha1
kind: RayCluster
metadata:
  name: two-sigma-ray-cluster
spec:
  headGroupSpec:
    replicas: 1
    template:
      spec:
        containers:
        - name: ray-head
          image: rayproject/ray:2.0.0
          resources:
            requests:
              cpu: "4"
              memory: "16Gi"
  workerGroupSpecs:
  - replicas: 2
    minReplicas: 2
    maxReplicas: 10
    template:
      spec:
        containers:
        - name: ray-worker
          image: rayproject/ray:2.0.0
          resources:
            requests:
              cpu: "2"
              memory: "8Gi"

# same Ray code, zero infra config
