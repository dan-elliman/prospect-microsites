# Tractian's use case: Distributed time-series feature engineering and batch inference with Ray
import ray
import random

ray.init()  # Connect to Anyscale

@ray.remote
def process_device_data(device_data):
    # Simulate feature engineering
    features = [x * 2 for x in range(100)]
    # Simulate inference
    inference_result = sum(features) * device_data
    return inference_result

fleet_data = [random.random() for _ in range(1000)]

# Distribute tasks across the cluster
results = ray.get([process_device_data.remote(device_data) for device_data in fleet_data])

print("Completed distributed processing on fleet.")  # scales to 100s of GPUs automatically

# Tractian's use case: Parallel time-series feature engineering and batch inference using Ray
import numpy as np
import pandas as pd
from sklearn.ensemble import RandomForestRegressor
import ray

# Initialize Ray
ray.init()

# Simulated data for multiple assets
assets = 100
timestamps = 1000

data = {f'asset_{i}': np.random.rand(timestamps) for i in range(assets)}

@ray.remote
def process_asset(asset_id, series):
    # Feature engineering
    features = np.array([series[i:i+10].mean() for i in range(len(series) - 10)])
    # Model inference
    model = RandomForestRegressor()
    model.fit(features.reshape(-1, 1), series[10:])
    prediction = model.predict(features.reshape(-1, 1))
    return asset_id, prediction

# Distribute tasks across the cluster
futures = [process_asset.remote(asset_id, series) for asset_id, series in data.items()]
results = ray.get(futures)

# This approach scales to 100s of GPUs automatically.

import ray
import numpy as np

# Simulate data
data = [np.random.rand(1000) for _ in range(100)]

@ray.remote
def process_series(series):
    feature = [series[i+1] - series[i] for i in range(len(series) - 1)]
    return sum(feature) / len(feature)  # Dummy inference

# Distributed feature engineering and inference
